Dedicated preset page for the shared re-auth/session-confirmation wrapper.

Locked Identifier Re-auth

const modal = createReauthFormModal({
  identifierValue: "operator@pbb.ph",
  fields: {
    identifier: "account_email",
    password: "reauth_password",
  },
  async onSubmit(values, ctx) {
    if (values.reauth_password !== "correct-password") {
      ctx.setErrors({ reauth_password: "Password does not match the active session." });
      return false;
    }
    return true;
  },
});
